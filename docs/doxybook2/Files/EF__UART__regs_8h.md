#ifndef CF_UARTREGS_H
#define CF_UARTREGS_H

#ifndef IO_TYPES
#define IO_TYPES
#define   __R     volatile const unsigned int
#define   __W     volatile       unsigned int
#define   __RW    volatile       unsigned int
#endif

#define CF_UART_CTRL_REG_EN_BIT 0
#define CF_UART_CTRL_REG_EN_MASK    0x1
#define CF_UART_CTRL_REG_TXEN_BIT   1
#define CF_UART_CTRL_REG_TXEN_MASK  0x2
#define CF_UART_CTRL_REG_RXEN_BIT   2
#define CF_UART_CTRL_REG_RXEN_MASK  0x4
#define CF_UART_CTRL_REG_LPEN_BIT   3
#define CF_UART_CTRL_REG_LPEN_MASK  0x8
#define CF_UART_CTRL_REG_GFEN_BIT   4
#define CF_UART_CTRL_REG_GFEN_MASK  0x10
#define CF_UART_CFG_REG_WLEN_BIT    0
#define CF_UART_CFG_REG_WLEN_MASK   0xf
#define CF_UART_CFG_REG_STP2_BIT    4
#define CF_UART_CFG_REG_STP2_MASK   0x10
#define CF_UART_CFG_REG_PARITY_BIT  5
#define CF_UART_CFG_REG_PARITY_MASK 0xe0
#define CF_UART_CFG_REG_TIMEOUT_BIT 8
#define CF_UART_CFG_REG_TIMEOUT_MASK    0x3f00
#define CF_UART_FIFOCTRL_REG_TXLT_BIT   0
#define CF_UART_FIFOCTRL_REG_TXLT_MASK  0xf
#define CF_UART_FIFOCTRL_REG_RXLT_BIT   8
#define CF_UART_FIFOCTRL_REG_RXLT_MASK  0xf00
#define CF_UART_FIFOS_REG_RXL_BIT   0
#define CF_UART_FIFOS_REG_RXL_MASK  0xf
#define CF_UART_FIFOS_REG_TXL_BIT   8
#define CF_UART_FIFOS_REG_TXL_MASK  0xf00

#define CF_UART_TXE_FLAG    0x1
#define CF_UART_RXF_FLAG    0x2
#define CF_UART_TXB_FLAG    0x4
#define CF_UART_RXA_FLAG    0x8
#define CF_UART_BRK_FLAG    0x10
#define CF_UART_MATCH_FLAG  0x20
#define CF_UART_FE_FLAG 0x40
#define CF_UART_PRE_FLAG    0x80
#define CF_UART_OR_FLAG 0x100
#define CF_UART_RTO_FLAG    0x200

typedef struct _CF_UART_TYPE_ {
    __R     RXDATA;
    __W     TXDATA;
    __W     PR;
    __W     CTRL;
    __W     CFG;
    __W     FIFOCTRL;
    __R     FIFOS;
    __W     MATCH;
    __R     reserved[952];
    __RW    im;
    __R     mis;
    __R     ris;
    __W     icr;
} CF_UART_TYPE;

#endif
